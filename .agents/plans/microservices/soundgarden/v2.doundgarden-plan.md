# PURPOSE

Soundgarden is the entry point of the media ingestion pipeline.

Responsibilities:

```
1. Accept authenticated user uploads
2. Validate audio files
3. Generate Track ID
4. Store upload temporarily
5. Emit ingestion events
```

Soundgarden **does NOT**:

```
transcode audio
perform AI analysis
approve tracks
store long-term media
serve playback
```

Those responsibilities belong to other services.

# USE CASE

### upload-track.use-case.ts

```ts
import { FileStoragePort } from "../ports/file-storage.port"
import { FileValidatorPort } from "../ports/file-validator.port"
import { EventBusPort } from "../ports/event-bus.port"
import { IdGeneratorPort } from "../ports/id-generator.port"

export class UploadTrackUseCase {

  constructor(
    private readonly storage: FileStoragePort,
    private readonly validator: FileValidatorPort,
    private readonly events: EventBusPort,
    private readonly ids: IdGeneratorPort
  ) {}

  async execute(file: Express.Multer.File) {

    const trackId = this.ids.generate()

    await this.events.publish("track.upload.received", {
      trackId,
      fileName: file.originalname
    })

    this.validator.validate(
      file.originalname,
      file.mimetype,
      file.size
    )

    await this.events.publish("track.upload.validated", {
      trackId
    })

    const path = await this.storage.store(
      trackId,
      file.originalname,
      file.buffer
    )

    await this.events.publish("track.upload.stored", {
      trackId,
      filePath: path
    })

    await this.events.publish("track.uploaded", {
      trackId,
      filePath: path
    })

    return { trackId }
  }
}
```

# STORAGE ADAPTER

### disk-storage.adapter.ts

```ts
import * as fs from "fs"
import * as path from "path"
import { FileStoragePort } from "../../application/ports/file-storage.port"

export class DiskStorageAdapter implements FileStoragePort {

  async store(
    trackId: string,
    fileName: string,
    buffer: Buffer
  ): Promise<string> {

    const dir = `/tmp/uploads/${trackId}`

    await fs.promises.mkdir(dir, { recursive: true })

    const filePath = path.join(dir, fileName)

    await fs.promises.writeFile(filePath, buffer)

    return filePath
  }
}
```

# VALIDATOR ADAPTER

### audio-validator.adapter.ts

```ts
import { FileValidatorPort } from "../../application/ports/file-validator.port"

export class AudioValidatorAdapter implements FileValidatorPort {

  validate(fileName: string, mimeType: string, fileSize: number) {

    const allowed = [
      "audio/mpeg",
      "audio/wav"
    ]

    const maxSize = 50 * 1024 * 1024

    if (!allowed.includes(mimeType)) {
      throw new Error("UNSUPPORTED_FORMAT")
    }

    if (fileSize > maxSize) {
      throw new Error("FILE_TOO_LARGE")
    }
  }
}
```

# HTTP CONTROLLER

### upload.controller.ts

```ts
import {
  Controller,
  Post,
  UploadedFile,
  UseInterceptors
} from "@nestjs/common"

import { FileInterceptor } from "@nestjs/platform-express"
import { UploadTrackUseCase } from "../../application/use-cases/upload-track.use-case"

@Controller("tracks")
export class UploadController {

  constructor(
    private readonly useCase: UploadTrackUseCase
  ) {}

  @Post("upload")
  @UseInterceptors(FileInterceptor("file"))
  async upload(@UploadedFile() file: Express.Multer.File) {

    return this.useCase.execute(file)
  }
}
```

import { NextRequest } from "next/server"

export async function POST(req: NextRequest) {

  const form = await req.formData()
  const soundgardenUrl = process.env.SOUNDGARDEN_BASE_URL ?? "http://localhost:7100"

  const response = await fetch(
    `${soundgardenUrl}/tracks/upload`,
    {
      method: "POST",
      body: form
    }
  )

  const json = await response.json()

  return Response.json(json)
}

#!/usr/bin/env bash
set -euo pipefail

BFF_URL="${BFF_URL:-http://localhost:3000}"
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
FILE="${SCRIPT_DIR}/data/remember-the-name.mp3"

echo "Uploading test track..."

response="$(curl -sS -w "\n%{http_code}" -X POST \
  -F "file=@${FILE}" \
  "${BFF_URL}/api/soundgarden/tracks/upload")"

http_code="$(echo "$response" | tail -n1)"
body="$(echo "$response" | sed '$d')"

if [[ "$http_code" != "200" ]]; then
  echo "Smoke test failed: expected 200, got $http_code"
  echo "$body"
  exit 1
fi

echo "Smoke test completed: $body"
